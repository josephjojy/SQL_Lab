DECLARE
a number;
n number;
s number := 0;
r number;
BEGIN
	a:= &a;
	n:=a;
	WHILE n>0 loop
		r:=mod(n,10);
		s:=(s*10)+r;
		n:=trunc(n/10);
	END loop;
	IF a=s THEN
		dbms_output.put_line('Number is a palindrome');
	else
		dbms_output.put_line('Number is not a palindrome');
	END IF;
END;
/
